let nums = [1, 2, 3, 4, 5, 6, 7];
let k = 3;

let rotate = function (nums, k) {
  nums.reverse();
  const len = nums.length;
  k %= len;

  const first = nums.slice(0, k).reverse();
  const second = nums.slice(k).reverse();

  const newNums = first.concat(second);

  for (let i = 0; i < len; i++) {
    nums[i] = newNums[i];
  }
  return nums;
};

console.log(rotate(nums, k));
